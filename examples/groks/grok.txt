function uid() {
return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function (c) {
var r = Math.random() * 16 | 0;
return (c === 'x' ? r : (r & 0x3 | 0x8)).toString(16);
});
}

function timeAgo(d) {
var s = Math.floor((Date.now() - d) / 1000);
if (s
< 60) return s + 's trước' ; if (s < 3600) return Math.floor(s / 60) + 'm trước' ; return Math.floor(s / 3600)
    + 'h trước' ; } function esc(s) { return String(s || '' ) .replace(/&/g,'&amp;').replace(/</g,'&lt;') .replace( />
/g,'&gt;').replace(/"/g,'&quot;');
}

function aspectClass(r) {
return { '16:9':'r169','1:1':'r11','2:3':'r23','3:2':'r32' }[r] || '';
}

function showError(msg) {
var el = document.getElementById('errorMsg');
if (!el) return;
el.textContent = msg;
el.style.display = 'block';
// Animate.css shake
el.classList.remove('animate__shakeX');
void el.offsetWidth; // reflow
el.classList.add('animate__shakeX');
setTimeout(function () { el.style.display = 'none'; }, 7000);
}

function updateStats() {
var t = state.videos.length;
var tp = document.getElementById('totalPill');
if (tp) tp.textContent = t + ' video' + (t !== 1 ? 's' : '');

var pp = document.getElementById('procPill');
if (pp) {
if (state.processing > 0) {
pp.style.setProperty('display','flex','important');
var pc = document.getElementById('procCount');
if (pc) pc.textContent = state.processing;
} else {
pp.style.setProperty('display','none','important');
}
}

if (em) em.style.display = state.videos.length === 0 ? 'flex' : 'none';

var keys = Object.keys(pollJobs).length;
if (pt) {
if (keys > 0) {
pt.classList.add('show');
var ptxt = document.getElementById('pollText');
if (ptxt) ptxt.textContent = 'Đang poll ' + keys + ' job(s)...';
} else {
pt.classList.remove('show');
}
}

function createCard(item) {
var c = document.createElement('div');
c.className = 'video-card animate__animated animate__fadeInUp';
c.id = 'card-' + item.id;
c.innerHTML = cardHTML(item);
return c;
}

function updateCard(item) {
var c = document.getElementById('card-' + item.id);
if (c) {
c.innerHTML = cardHTML(item);
// pulse on done
if (item.status === 'done') {
c.classList.add('animate__animated','animate__pulse');
setTimeout(function () { c.classList.remove('animate__pulse'); }, 800);
}
}

    (!item) return; item.progress=pct; var pe=document.getElementById('pct-' + id); var
    fe=document.getElementById('fill-' + id); var se=document.getElementById('sub-' + id); if (pe) pe.textContent=pct
    + '%' ; if (fe) fe.style.width=pct + '%' ; if (se) se.textContent=sub || '' ; } /* ═══════════════════════════════
    POLLING ═══════════════════════════════ */ function startPolling(guildId, expectedCount, onResult) { var
    MAX_MS=300000; var job={ guildId: guildId, expectedCount: expectedCount, renderedCount: 0, interval: null,
    startTime: Date.now(), onResult: onResult }; job.interval=setInterval(function () { // Timeout if (Date.now() -
    job.startTime> MAX_MS) {
    clearInterval(job.interval); delete pollJobs[guildId];
    state.videos.forEach(function (v) {
    if (v.guildId === guildId && v.status === 'generating') {
    v.status = 'error'; v.errorMsg = 'Timeout 300s: không nhận được link video';
    state.processing = Math.max(0, state.processing - 1);
    updateCard(v);
    }
    });
    updateStats(); return;
    }

    try {
    var live = (window.__ac && window.__ac.live) ? window.__ac.live : {};
    var res = live[guildId];
    if (res && Array.isArray(res)) {
    var newItems = res.slice(job.renderedCount);
    if (newItems.length > 0) {
    job.renderedCount += newItems.length;
    onResult(newItems, job.renderedCount, expectedCount);
    }
    if (job.renderedCount >= expectedCount) {
    clearInterval(job.interval); delete pollJobs[guildId]; updateStats();
    }
    }
    } catch (_) {}
    }, 1000);

    pollJobs[guildId] = job;
    updateStats();
    }
